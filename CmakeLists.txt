# Set the minimum version of CMake required to build this project
cmake_minimum_required(VERSION 3.29.3)
# Define the project name, version, and languages used
project(RiftSpace
      VERSION 1.0.0
      LANGUAGES C CXX
)
# Use modern C++14 standard
set(CMAKE_CXX_STANDARD          14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS       OFF)


# Include FetchContent module to download SFML directly
include(FetchContent)
# Define a variable to hold the SFML library name
set(SFML_LIB_NAME SFML)
# Tell CMake to fetch SFML from GitHub (v2.6.0)
FetchContent_Declare(
    ${SFML_LIB_NAME}
    GIT_REPOSITORY https://github.com/SFML/SFML
    GIT_TAG 2.6.0
    GIT_SHALLOW FALSE
)
# Make the SFML library available to our project
FetchContent_MakeAvailable(${SFML_LIB_NAME})


# Define a variable for our target (game executable) name
set(RIFT_SPACE_GAME_TARGET_NAME RiftSpace)
# Add the RiftSpace/ subdirectory where another CMakeLists.txt exists
add_subdirectory(RiftSpace)
